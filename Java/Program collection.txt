//Reverse a sentence word by word

public class Main
{
    public static void main(String[] args) {
    
        Scanner in = new Scanner(System.in);
        String input = in.nextLine();
	    String revString = "";
	    String[] words = input.split("\\s");

	    for(int i=0;i<words.length;i++){
	       String word =  words[i];
	       String revWord = "";
	       for(int j=word.length()-1;j>=0;j--){
	        revWord+=word.charAt(j);
	       }
	        revString+=revWord+" ";
	    }
	    System.out.println(revString);
	}
}

//Find a square root of a number using Newton's method
import java.util.Scanner;
import java.lang.Math;
public class Main
{
	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
        Float Number = in.nextFloat();
        float sqrRoot = 0;
        float temp = 0;
        temp = Number/2;
        while (Number >= 0.0)    {
            sqrRoot = temp - (temp * temp - Number) / (2 * temp);
            float value;
            value = Math.abs(temp - sqrRoot);
            if (value < .0001)
                break; 
            else temp = sqrRoot;
        } 
        System.out.println(sqrRoot);
	}
}

//remove duplicates from an array in place
import java.util.Arrays;
public class DemoApplication{
	public static void main(String[] args) {
		int[] test = new int[]{1, 1, 2, 2, 3, 4, 5};
		System.out.println("Array with Duplicates       : " + Arrays.toString(test));
        System.out.println("After removing duplicates   : " + Arrays.toString(removeDuplicates(test)));
    }
	public static int[] removeDuplicates(int[] numbersWithDuplicates) {
        Arrays.sort(numbersWithDuplicates);     
        int[] result = new int[numbersWithDuplicates.length];
        int previous = numbersWithDuplicates[0];
        result[0] = previous;

        for (int i = 1; i < numbersWithDuplicates.length; i++) {
            int ch = numbersWithDuplicates[i];

            if (previous != ch) {
                result[i] = ch;
            }
            previous = ch;
        }
        return result;
    }
}

//Reverse an array irrespective of the datatype
import java.util.*;
public class Main
{
	/* Function to reverse arr[] from start to end*/
    static void revereseArray(Object arr[]) 
    { 
        Object temp; 
        int start = 0;
        int end = arr.length-1;
        while (start < end) 
        { 
            temp = arr[start];  
            arr[start] = arr[end]; 
            arr[end] = temp; 
            start++; 
            end--; 
        }  
    }      
       
    /* Utility that prints out an array on a line */
    static void printArray(Object arr[]) 
    { 
        for (int i = 0; i < arr.length; i++) 
             System.out.print(arr[i] + " "); 
           
         System.out.println(); 
    }  
  
    // Driver code 
    public static void main(String args[]) { 
          
        Object arr[] = {'a', 2, 3, 4, 5, 6, 'g'}; 
        printArray(arr); 
        revereseArray(arr); 
        System.out.print("Reversed array is \n"); 
        printArray(arr);  
         
    } 
}
